# import "random" module to generate/shuffle random elements from a list
import random

#lists for selecting/shuffling the elements to generate a random passowrd
lower_case = "abcdefghijklmnopqrstuvwxyz"
upper_case = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
numbers = "0123456789"
symbols = "!@#$%^&*|\/.,"

mixture = lower_case + upper_case + numbers + symbols
length = 8

password = "".join(random.sample(mixture, length))
print(password)
